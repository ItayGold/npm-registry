import { Component } from '@angular/core';

@Component({
  selector: 'click-date-picker-page',
  templateUrl: './date-picker-page.component.html',
  styleUrls: ['./date-picker-page.component.scss'],
})
export class DatePickerPageComponent {
  selectedDate: Date = new Date();
  todayDate: Date = new Date();
  todayButtonText: string = String(`Today`);

  setDateEvent($event: Date): void {
    const date = this.formatDate($event);
  }

  private formatDate(date: Date): string {
    const d = new Date(date);
    const year = d.getFullYear();
    let month = '' + (d.getMonth() + 1);
    let day = '' + d.getDate();

    if (month.length < 2) {
      month = '0' + month;
    }
    if (day.length < 2) {
      day = '0' + day;
    }

    return [year, month, day].join('-');
  }
}
